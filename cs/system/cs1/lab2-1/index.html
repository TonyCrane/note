    module CarryLookaheadAdder4(
    input [3:0] A,
    input [3:0] B,
    input Cin,
    output [3:0] S,
    output P,
    output G,
    output Cout
);
    wire [3:0] p;
    wire [3:0] g;
    assign p = A ^ B;
    assign g = A & B;

    wire [4:0] c;
    assign c[0] = Cin;
    assign c[1] = g[0] | (p[0] & c[0]);
    assign c[2] = g[1] | (p[1] & g[0]) | (p[1] & p[0] & c[0]);
